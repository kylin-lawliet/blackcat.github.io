    下拉框加图标显示

    <input id="iconCls"  name="iconCls" class="easyui-combobox" />

var data=comboboxData(text);
$('#iconCls').combobox( {
    showItemIcon:true,// 显示图标
    panelHeight: 'auto',//自适应
    valueField: 'value',//绑定字段value
    textField: 'text',//绑定字段text
    onLoadSuccess:function(){
        $(".combo").click(function(){
            $(this).prev().combobox("showPanel");
        });
    },
    data:data,
    formatter: function (row) {
        var opts = $(this).combobox('options');
        return row[opts.textField]
    }
});

    下拉框添加多选

    <input id="ceckbox" class="easyui-combobox" />

var data=comboboxData(text);
$('#ceckbox').combobox( {
    multiple: true,// 支持多选
    panelHeight: 'auto',//自适应
    valueField: 'value',//绑定字段value
    textField: 'text',//绑定字段text
    onLoadSuccess:function(){
        $(".combo").click(function(){
            $(this).prev().combobox("showPanel");
        });
    },
    data:[data],
    formatter: function (row) {
        var opts = $(this).combobox('options');
        return '<input type="checkbox" class="combobox-checkbox" style="margin:0 5px;vertical-align: -2px" id="' + row[opts.valueField] + '">' + row[opts.textField]
    },
    onSelect: function (row) {// 选择树节点触发事件
        var opts = $(this).combobox('options');
        var el = opts.finder.getEl(this, row[opts.valueField]);
        el.find('input.combobox-checkbox')._propAttr('checked', true);
    },
    onUnselect: function (row) {// 取消选择checkbox
        var opts = $(this).combobox('options');
        var el = opts.finder.getEl(this, row[opts.valueField]);
        el.find('input.combobox-checkbox')._propAttr('checked', false);
    }
});

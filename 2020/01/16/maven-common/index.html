    其他

    java持久化API

    java持久化API，就是类似与ORM的东西，主要用于和数据库交互的，比如Java bean和数据库表的对应关系，这个是JPA的核心包，属于Java EE 标准。
    如：@Entity和@Table，@Transient注解

    <!--java持久化API-->
<dependency>
    <groupId>javax.persistence</groupId>
    <artifactId>persistence-api</artifactId>
    <version>1.0</version>
</dependency>

    AOP依赖

    <!-- AOP依赖,一定要加,否则权限拦截验证不生效 -->
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-aop</artifactId>
</dependency>

    maven打包

    <plugin>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-maven-plugin</artifactId>
    <!-- 当所依赖的其他模块，有启动类的时候，需要以下配置，指定唯一启动类，
            本次事例未涉及其他模块有启动类，可自行测试 -->
    <configuration>
        <!-- 指定该Main Class为全局的唯一入口 -->
        <mainClass>com.blackcat.blog.BlogBaseApplication</mainClass>
        <layout>ZIP</layout>
    </configuration>
    <executions>
        <execution>
            <goals>
                <goal>repackage</goal><!--可以把依赖的包都打包到生成的Jar包中-->
            </goals>
        </execution>
    </executions>
</plugin>
